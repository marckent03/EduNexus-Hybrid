  <script>
    let currentStep = 1;
    let selectedRole = 'student';

    window.addEventListener('scroll', () => {
      document.getElementById('navbar').classList.toggle('scrolled', window.scrollY > 10);
    });

    function togglePw(id) {
      const el = document.getElementById(id);
      el.type = el.type === 'password' ? 'text' : 'password';
    }

    function checkStrength(val) {
      const segs = [document.getElementById('s1'), document.getElementById('s2'),
                    document.getElementById('s3'), document.getElementById('s4')];
      const lbl  = document.getElementById('strengthLabel');
      const colors = ['#ef4444','#f59e0b','#10b981','#3730a3'];
      const labels = ['Weak','Fair','Good','Strong'];
      segs.forEach(s => s.style.background = 'var(--border)');
      let score = 0;
      if (val.length >= 8) score++;
      if (/[A-Z]/.test(val)) score++;
      if (/[0-9]/.test(val)) score++;
      if (/[^A-Za-z0-9]/.test(val)) score++;
      for (let i = 0; i < score; i++) segs[i].style.background = colors[Math.min(score-1,3)];
      lbl.textContent = score > 0 ? labels[Math.min(score-1,3)] : 'Password strength';
      lbl.style.color = score > 0 ? colors[Math.min(score-1,3)] : 'var(--text-500)';
    }

    function goStep(n) {
      document.getElementById('step'+currentStep).style.display = 'none';
      document.getElementById('step'+currentStep+'-tab').classList.remove('active');
      if (n > currentStep) {
        document.getElementById('step'+currentStep+'-tab').classList.add('done');
        document.getElementById('step'+currentStep+'-tab').querySelector('.step-dot').textContent = '✓';
        const line = document.getElementById('line'+currentStep);
        if (line) line.classList.add('done');
      } else {
        document.getElementById('step'+currentStep+'-tab').classList.remove('done');
        document.getElementById('step'+currentStep+'-tab').querySelector('.step-dot').textContent = currentStep;
        const line = document.getElementById('line'+(currentStep-1));
        if (line) line.classList.remove('done');
      }
      currentStep = n;
      document.getElementById('step'+currentStep).style.display = 'block';
      document.getElementById('step'+currentStep+'-tab').classList.add('active');
      document.getElementById('step'+currentStep+'-tab').classList.remove('done');
      document.getElementById('step'+currentStep+'-tab').querySelector('.step-dot').textContent = currentStep;
    }

    function selectRole(el) {
      document.querySelectorAll('.role-card').forEach(c => c.classList.remove('selected'));
      el.classList.add('selected');
    }

    function selectRegRole(el, role) {
      document.querySelectorAll('#reg-role-grid .role-card').forEach(c => c.classList.remove('selected'));
      el.classList.add('selected');
      selectedRole = role;
    }

    }
