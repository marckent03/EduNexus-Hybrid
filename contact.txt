  <script>
    window.addEventListener('scroll', () => {
      document.getElementById('navbar').classList.toggle('scrolled', window.scrollY > 10);
    });

    function setPriority(level) {
      const styles = {
        low:    { border: 'var(--success)', bg: '#d1fae5', color: '#065f46' },
        medium: { border: 'var(--primary)', bg: 'var(--primary-pale)', color: 'var(--primary)' },
        high:   { border: 'var(--danger)',  bg: '#fee2e2', color: '#991b1b' },
      };
      ['low','medium','high'].forEach(k => {
        const el = document.getElementById('p' + k.charAt(0).toUpperCase() + k.slice(1));
        if (k === level) {
          el.style.borderColor = styles[k].border;
          el.style.background  = styles[k].bg;
          el.style.color       = styles[k].color;
        } else {
          el.style.borderColor = 'var(--border)';
          el.style.background  = 'transparent';
          el.style.color       = 'var(--text-700)';
        }
      });
    }

    function toggleFaq(item) {
      const answer  = item.querySelector('.faq-a');
      const toggle  = item.querySelector('.faq-toggle');
      const isOpen  = item.classList.contains('open');
      document.querySelectorAll('.faq-item').forEach(f => {
        f.classList.remove('open');
        f.querySelector('.faq-a').classList.remove('open');
      });
      if (!isOpen) {
        item.classList.add('open');
        answer.classList.add('open');
      }
    }

    document.getElementById('contactForm').addEventListener('submit', function(e) {
      e.preventDefault();
      const btn = document.getElementById('submitBtn');
      btn.textContent = 'Sending…';
      btn.disabled = true;
      setTimeout(() => {
        document.getElementById('successMsg').style.display = 'flex';
        btn.innerHTML = '✓ Message Sent';
        btn.style.background = 'var(--success)';
      }, 1200);
    });
  </script>
